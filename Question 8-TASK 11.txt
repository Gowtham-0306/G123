package leaning.selenium;

import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;

public class FileReadingProgram {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter the file path: ");
        String filePath = scanner.nextLine();

        try {
            readFileData(filePath);
            //  logic to process the  file data
        } catch (FileNotFoundException e) {
            System.out.println("File not found: " + e.getMessage());
            //  logic to handle a file - if not found
        }
    }

    private static void readFileData(String filePath) throws FileNotFoundException {
        File file = new File(filePath);

        // to check the file exist or not
        if (!file.exists()) {
            throw new FileNotFoundException("The mentioned file does not exist.");
        }

        // If the file exist below will read 
        try (Scanner fileScanner = new Scanner(file)) {
            while (fileScanner.hasNextLine()) {
                String line = fileScanner.nextLine();
                System.out.println(line); 
            }
        }
    }
}
