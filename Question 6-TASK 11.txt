package leaning.selenium;

import java.util.Scanner;

public class IncorrectPasswordLogin {

    private static final String CORRECT_PASSWORD = "correct_password";

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter your password: ");
        String enteredPassword = scanner.nextLine();

        try {
            login(enteredPassword);
            System.out.println("Login successful!");
            // Additnal  logic for successful login 
        } catch (IncorrectPasswordException e) {
            System.out.println("Login failed: " + e.getMessage());
            // Additional logic for handling incorrect password can go here
        }
    }

    private static void login(String enteredPassword) throws IncorrectPasswordException {
        // Check if the entered password os correct
        if (!enteredPassword.equals(CORRECT_PASSWORD)) {
            throw new IncorrectPasswordException("Incorrect password. Please try again.");
        }
        // If passwords is correct , continue with the login process
    }

    // Custom exception class for incorrect password
    static class IncorrectPasswordException extends Exception {
        public IncorrectPasswordException(String message) {
            super(message);
        }
    }
}
