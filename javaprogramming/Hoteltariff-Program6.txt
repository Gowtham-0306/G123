package learning.java;

import java.util.Scanner;

public class HotelTariff {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.println("Enter the number of the month here : ");
        System.out.println(" (FOR EX JAN = 1 and Dec = 12) ");
        int month = scanner.nextInt();
        
        System.out.print("Enter the room rent per day: ");
        double roomRent = scanner.nextDouble();
        
        System.out.print("Enter the number of days stayed: ");
        int daysStayed = scanner.nextInt();

        double totalTariff = calculateTariff(month, roomRent, daysStayed);
        System.out.printf("Hotel Tariff: %.2f%n", totalTariff);

        scanner.close();
    }

    public static double calculateTariff(int month, double roomRent, int daysStayed) {
        double totalTariff = roomRent * daysStayed;

        // Check if the month is during the peak month by using the switch construct
        switch (month) {
            case 4:
            case 5:
            case 6:
            case 11:
            case 12:
                totalTariff += (0.20 * totalTariff); // if the months falls uder te peak month we are adding extra Add 20% during peak season
                break;
        }

        return totalTariff;
    }
}
